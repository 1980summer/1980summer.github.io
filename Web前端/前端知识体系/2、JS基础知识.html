现在的需求是，让obj2 等于 obj1 的时候，obj2改了但obj1 不变，那么就要用到深拷贝，代码如下

​ （注意：代码里面要判断基本数据类型和引用数据类型、判断是数组还是对象、递归技术的使用）

        const obj1 = {
            name: 'Alice',
            age:20,
            address: {
                city: 'Paris'
            },
            arr:['a', 'b', 'c']
        }
        // 定义一个空数组
        const obj = {}
        // 定义深拷贝函数
        function deepClone ( obj ) {
            // 判断被拷贝的，如果不是对象也不是数组，或者是一个空对象，就直接返回
            // obj是基本数据类型则走这条逻辑
            if ( typeof obj !== 'object' || obj == null ) return obj
            // 定义一个新的变量，用来保存返回结果
            let result
            // 对象或者数组走这条逻辑
            if ( obj instanceof Array ) {
                result = []
            } else {
                result = {}
            }
            // 不论是对象还是数组，都能用 for in 来遍历属性
            for (let k in obj ) {
                //  保证 k 不是原型的属性
                // hasOwnProperty表示是否有自己的属性。这个方法会查找一个对象是否有某个属性，但是不会去查找它的原型链。
                if ( obj.hasOwnProperty(k) ) {
                    result[k] = deepClone(obj[k])
                }
            }
            // 返回结果
            return result
        }

        // 对 obj1 进行深拷贝
        const obj2 = deepClone(obj1)
        obj2.address.city = 'New-York'
        console.log(obj1.address.city);  // 仍然返回 Paris

# 2、class的原型本质，怎么理解？

*A：*可以通过下面这段代码，把表默写出来。

        // 父类
        class People {
            constructor(name) {
                this.name = name
            }
            eat() {
                console.log(`${this.name} eat something`)
            }
        }
        // 子类
        class Student extends People {
            constructor(name, number) {
                super(name)
                this.number = number
            }
            sayHi() {
                console.log(`姓名 ${this.name} 学号 ${this.number}`)
            }
        }
        // 子类
        class Teacher extends People {
            constructor(name, major) {
                super(name)
                this.major = major
            }
            teach() {
                console.log(`${this.name} 教授 ${this.major}`)
            }
        }
        // 实例
        const xialuo = new Student('夏洛', 100)
        console.log(xialuo.name)
        console.log(xialuo.number)
        xialuo.sayHi()
        xialuo.eat()

        // 实例
        const wanglaoshi = new Teacher('王老师', '语文')
        console.log(wanglaoshi.name)
        console.log(wanglaoshi.major)
        wanglaoshi.teach()
        wanglaoshi.eat()

# 3、手写一个简单的 jQuery，考虑插件和扩展性

（通过手写一个简易的DOM节点查询功能的 jQuery 实例，了解 jQuery 内部实现原理及理解Class与继承。

结合class和继承来理解

class jQuery {
    constructor(selector) {
        const result = document.querySelectorAll(selector)
        const length = result.length
        for (let i = 0; i < length; i++) {
            this[i] = result[i]
        }
        this.length = length
        this.selector = selector
    }
    get(index) {
        return this[index]
    }
    each(fn) {
        for (let i = 0; i < this.length; i++) {
            const elem = this[i]
            fn(elem)
        }
    }
    on(type, fn) {
        return this.each(elem => {
            elem.addEventListener(type, fn, false)
        })
    }
    // 扩展很多 DOM API
}
