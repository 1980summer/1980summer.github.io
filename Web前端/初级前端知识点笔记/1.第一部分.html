# 前端初中级面试题 (Part 1)

# 1、var、let、const的区别

  • var是 ES5语法

  • let const是 ES6语法;var有变量提升，

  • var和let是变量，可修改;const是常量，不可修改;

  • let、const 有块级作用域， var 没有

  //  变量提升 ES5
  console.log(a) // undefined
  var a = 200
   
  let b
  console.log(b) // 报错
  b = 200
  // 块级作用域
  // var定义的变量
  if (true) {
      var name = 'Matt';
      console.log(name); // Matt
  }
  console.log(name); // Matt
  
  // let定义的变量
  if (true) {
      let age = 26;
      console.log(age); // 26
  }
  console.log(age); // ReferenceError: age没有定义

# 4、手写深度比较，模拟lodash isEqual （判断两个数据是否相等，模拟 ===, ）

注意：Object.keys() 方法会返回一个由一个给定对象的自身可枚举属性组成的数组

// 判断是否是对象或数组
function isObject(obj) {
    return typeof obj === 'object' && obj !== null
}
// 全相等（深度，意思就是对象里还有对象）
function isEqual(obj1, obj2) {
    // 如果有其中一个不是 Object 对象的话
    if (!isObject(obj1) || !isObject(obj2)) {
        // 值类型（注意，参与 equal 的一般不会是函数）
        return obj1 === obj2  // 这个  obj1 === obj2里面 既有可能是 true 也有可能是 false, 如果是false就直接返回了, 如果是 true 才继续往下判断
    }
    // 假如有人故意传入两个相等的对象，如 isEqual(obj1, obj1)
    if (obj1 === obj2) {
        return true
    }
    // 两个都是对象或数组，而且不相等
    // 1. 先取出 obj1 和 obj2 的 keys ，比较个数
    const obj1Keys = Object.keys(obj1)
    const obj2Keys = Object.keys(obj2)
    // keys 的个数不相等那肯定不全等
    if (obj1Keys.length !== obj2Keys.length) {
        return false
    }
    // 2. keys 的个数相等,也不一定就全等,继续往下判断
    // 以 obj1 为基准，和 obj2 依次递归比较
    for (let key in obj1) { // 对 obj1 进行遍历
        // 比较当前 key 的 val —— 递归！！！
        const res = isEqual(obj1[key], obj2[key])
        if (!res) {
            return false
        }
    }
    // 3. for in 循环全部遍历完, 并且没有命中第2步的false的话,那就表明两个对象全相等
    return true
}

// 测试
const obj1 = {
    a: 100,
    b: {
        x: 100,
        y: 200
    }
}
const obj2 = {
    a: 100,
    b: {
        x: 100,
        y: 200
    }
}
// console.log( obj1 === obj2 )
console.log( isEqual(obj1, obj2) )

# 6、数组的pop push unshift shift 分别做什么

  • 功能是什么?
  • 返回值是什么?
  • 是否会对原数组造成影响?
方法 功能 返回值 对原数组的影响
pop（） 移除数组最后一项 被移除的那一项 修改原数组
push（） 在原数组末尾加上括号里的元素 length 修改原数组
unshift（） 将参数添加到原数组开头 length 修改原数组
shift ( ) 移除数组第一项 被移除的那一项 修改原数组

# 8、slice 和 splice 的区别？

  • 功能区别( slice -切片，splice -剪接)
  • 参数和返回值
  • 是否纯函数?
const arr = [10, 20, 30, 40, 50]

// // slice 纯函数

const arr1 = arr.slice()  // 新的[10, 20, 30, 40, 50] （内存地址和原数组不一样）
const arr2 = arr.slice(1, 4) // [20, 30, 40,]
const arr3 = arr.slice(2) // [30, 40, 50]
const arr4 = arr.slice(-3) // [30, 40, 50]

// // splice 非纯函数 剪接剪接，剪掉再接上

const spliceRes = arr.splice(1, 2, 'a', 'b', 'c')  
const spliceRes1 = arr.splice(1, 2)
const spliceRes2 = arr.splice(1, 0, 'a', 'b', 'c')
console.log(spliceRes, arr)
